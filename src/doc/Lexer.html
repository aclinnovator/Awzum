def initialize


        @token_operations = {}
        @indent = 0
        @indent_stack = []
        #By storing the various reactions to regexs in a dictionary, it makes expanding the parser much easier and is one level below building a custom parser
        @token_operations[:id] = lambda do |match, array|
                if KEYWORDS.include? match
                        array << Token.new(match.upcase.to_sym, match)
                end  
                 match.size
        end
        
        @token_operations[:constant] = lambda do |match, array|
                array << Token.new(match.upcase.to_sym, match)
                match.size
        end

        @token_operations[:number] = lambda do |match,array|
                array << Token.new(match.upcase.to_sym, match)
                match.size                   
        end

        @token_operations[:string] = lambda do |match, array|
                array << Token.new(match.upcase.to_sym, match)
                match.size+2         
        end
        
        @token_operations[:indent_tabs] = lambda do |match,array|
                raise "Indent level inconsistent. Got #{match.size} >> expected #{@indent}" if match.size < @indent
                @indent +=match.size
                @indent_stack.push @indent 
                array << Token.new(match.upcase.to_sym, match)
                match.size+2
        end
        @token_operations[:indent_spaces] = lambda do |match, array|
                if match.size == @indent 
      array << [:NEWLINE, "\n"] # Nothing to do, we're still in the same block
    elsif match.size < @indent # Case 3
      while match.size < @indent
        @indent_stack.pop
        @indent = @indent_stack.last || 0
        array << [:DEDENT, match.size]
      end
      array << [:NEWLINE, "\n"]
    else # indent.size > current_indent, error!
      raise "Missing ':'" # Cannot increase indent level without using ":"
    end
    return match.size + 1
        end

        @token_operations[:operator] = lambda do |match,array|
                array << [match.to_sym, match]
                match.size
        end

        @token_operations[:space] = lambda do |match,array|
                1 #Will increment the current index in the code by 1
        end

        @token_operations[:other] = lambda do |match,array|
   array << [match, match]
   1 #Handling other operators
        end
        raise "Token operations is not fitted to all availbale tokens:
               Ops   : #{@token_operations.keys},
               Tokens: #{TOKENS.keys}" unless (@token_operations.keys.to_set.subset? TOKENS.keys.to_set) 
end

def match_first string, hash
        keys = hash.keys
        for key in keys
                value = hash[keys]
                match = string[value,1]
                return Struct.new(string:match,key:key] if match 
        end
end

def tokenize code
        code.chomp! #Removing all newline characters and other ugly parts of the string we dont really like. 
        tokens = []
        @indent = 0
        @indent_stack = []
        i = 0 #Current index of code being read
        while i < code.size #run through entire code
                codon = code[i..-1]
                match_data = match_first codon, TOKENS
                i+= @token_operations match_data[]
                while @indent = @indent_stack.pop #Close any blocks that end without a dedent
                tokens << [:DEDENT, @indent_stack.first || 0]
        end 
        end
        tokens
end
